	//title 속성을 가지고 있는 태그의 배경색 변경
	$('p[title]').css('background','yellow');
	
	//title 값이 금요일인 요소를 선택해서 글자색을 변경한다
	$('p[title=금요일]').css('color','red');
	
	//id가 p글자로 시작하는 요소를 선택
	$('p[id^=p]').css('font-size','1.5em');
	
	//id가 3으로 끝나는 요소의 테두리
	$('p[id$=3]').css('border','3px double green');
	
	//class가 pp가 아닌 요소를 선택하여 밑줄
	$('p[class!=pp]').css('text-decoration',"underline");
	
	//title에 love 글자가 포함되어 있는 요소를 선택
	$('p[title*=love]').css('color','green');
	
	//title에 love 단어가 포함되어 있는 요소를 선택
	$('p[title~=love]').css('background','cyan')
	
	//class가 aa이거나 aa-로 시작하는 요소를 선택
	$('p[class |= aa]').css('font-weight','bold')
