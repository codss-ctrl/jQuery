이벤트 등록 방법 - delegate()
이미 만들어져  버튼 class=btn 이나 
실행 후에 동적으로 새롭게 만들어진 버튼 class=btn에
동시에 같은 이벤트 처리기를 등록한다.

$('부모요소 또는 document').delegate('이벤트대상요소', '이벤트이름', function(){})
$('부모요소 또는 document').on('이벤트이름','이벤트대상요소',function(){})
$('.dd').on('click', '.btn', function(){})
$(document).on('click','.btn',function(){})

처음부터 있었던 .btn에만 이벤트 적용  - bind
$('.btn').on('click',function(){})

처음부터 있었던 .btn 뿐만 하니라 실행 후에 생성된 btn에도 같은 이벤트가 실행된다. - delegate
$('.dd').on('click', '.btn', function(){})
